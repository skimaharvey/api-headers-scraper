#calculate median 
def median(array)
  return nil if array.empty?
  sorted = array.sort
  len = sorted.length
  (sorted[(len - 1) / 2] + sorted[len / 2]) / 2.0
end

#calculate average 
def average(arr)
arr.inject{ |sum, el| sum + el }.to_f / arr.size
end
